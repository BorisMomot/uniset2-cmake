set(IDL_PROCESSES_LIST IOController_i.idl)

set(IDL_PROCESSES_DIR ${CMAKE_CURRENT_SOURCE_DIR})
set(CPP_PROCESSES_DIR ${CMAKE_CURRENT_BINARY_DIR})

foreach(IDL_FILE ${IDL_PROCESSES_LIST})
  execute_process(
    COMMAND omniidl -bcxx ${IDL_PROCESSES_DIR}/${IDL_FILE} 
    WORKING_DIRECTORY ${CPP_PROCESSES_DIR}
    )
endforeach()

source_group(TREE "${CMAKE_CURRENT_SOURCE_DIR}" FILES ${CPP_PROCESSES_DIR})
